class

class User {

  constructor(name) {
    this.name = name;
  }
  
  sayHi() {
    alert(this.name);
  }

}

let user = new User("Henadzi");
user.sayHi(); // Henadzi

function extend(Child, Parent) {
    Child.prototype = Object.create(Parent.prototype);
    Child.prototype.constructor = Child;
    Child.superclass = Parent.prototype; // optional
}
